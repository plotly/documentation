GGPLOT - geom_point

Create scatterplots and then convert them with ggplotly

p <- ggplot(mtcars, aes(wt, mpg))
p <-  p + geom_point()

plotly::ggplotly(p)

p <- ggplot(mtcars, aes(wt, mpg))
p <-  p + geom_point(aes(colour = factor(cyl)))

plotly::ggplotly(p)

p <- ggplot(mtcars, aes(wt, mpg))
p <-  p + geom_point(aes(shape = factor(cyl)))

plotly::ggplotly(p)
