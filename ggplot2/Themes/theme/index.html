GGPLOT - theme

Give plots a consistent customized look and then convert them with ggplotly.

p1 <- ggplot(mtcars, aes(wt, mpg)) +
  geom_point() +
  labs(title = "Fuel economy declines as weight increases")

plotly::ggplotly(p1)

p1 <- ggplot(mtcars, aes(wt, mpg)) +
  geom_point() +
  labs(title = "Fuel economy declines as weight increases")
p <-  p1 + theme(plot.title = element_text(size = rel(2)))

plotly::ggplotly(p)
