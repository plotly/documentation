library(plotly) x <- rep(1:10, 2) y <- c(1:10, 1:10+5) fac <- gl(2, 10) df <- data.frame(x=x, y=y, fac=fac) ggplot(df, aes(x=x, y=y, linetype=fac)) + geom_line() + geom_segment(aes(x=2, y=7, xend=7, yend=7), colour="red") + scale_linetype_discrete(guide=guide_legend(override.aes=aes(colour="blue"))) ggplotly()
