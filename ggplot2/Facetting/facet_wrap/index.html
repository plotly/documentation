GGPLOT - facet_wrap

Wraps a 1d sequence of panels into 2d and then convert them with ggplotly.

p <- ggplot(mpg, aes(displ, hwy)) + geom_point()
p <-  p + facet_wrap(vars(class))

plotly::ggplotly(p)

p <- ggplot(mpg, aes(displ, hwy)) + geom_point()
p <-  p + facet_wrap(vars(class), nrow = 4)

plotly::ggplotly(p)
