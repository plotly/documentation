import datetime import matplotlib.pyplot as plt import numpy as np import plotly.plotly as py import plotly.tools as tls # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api x = np.array([datetime.datetime(2014, i, 9) for i in range(1,13)]) y = np.random.randint(100, size=x.shape) fig = plt.figure() ax1 = fig.add_subplot(111) ax1.plot(x,y) ax1.set_title('Setting Custom Axis Range for time series') plotly_fig = tls.mpl_to_plotly( fig ) plotly_fig['layout']['xaxis1']['range'] = [1357669800000, 1449599400000] plotly_url = py.plot(plotly_fig, filename='mpl-time-series-custom-axis')
