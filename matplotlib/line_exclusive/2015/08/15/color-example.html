# Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api import matplotlib.pyplot as plt import matplotlib as mpl import numpy as np import plotly.plotly as py mpl_color = plt.figure() # Generate data... nx, nsteps = 100, 20 x = np.linspace(0, 1, nx) data = np.random.random((nx, nsteps)) - 0.5 data = data.cumsum(axis=0) data = data.cumsum(axis=1) # Plot cmap = mpl.cm.autumn for i, y in enumerate(data.T): plt.plot(x, y, color=cmap(i / float(nsteps))) plot_url = py.plot_mpl(mpl_color, filename='mpl-color-example')
