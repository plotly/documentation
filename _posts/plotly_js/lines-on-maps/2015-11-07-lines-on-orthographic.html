--- name: Lines on an Orthographic Map plot_url: https://plot.ly/~jackp/9098/ language: plotly_js suite: lines-on-maps order: 1 sitemap: false --- function scattergeolines('myDiv') { var colors = ['rgb(213,62,79)', 'rgb(254,224,139)', 'rgb(102,194,165)'], lon = _.range(-180, 180, 0.5), data = []; for(var i = 0; i < colors.length; i++) { data.push({ type: 'scattergeo', mode: 'lines', lon: lon, lat: _.map(lon, function(loni) { return 15*(i+1) * Math.sin(4 * loni * Math.PI / 180) + 15; }), line: { color: colors[i] } }); } var layout = { geo: { projection: { type: 'orthographic', rotation: { lon: -100, lat: 40 }, }, showocean: true, oceancolor: 'rgb(0, 255, 255)', showland: true, landcolor: 'rgb(230, 145, 56)', showlakes: true, lakecolor: 'rgb(0, 255, 255)', showcountries: true, lonaxis: { showgrid: true, gridcolor: 'rgb(102, 102, 102)' }, lataxis: { showgrid: true, gridcolor: 'rgb(102, 102, 102)' } } }; Plotly.newPlot('myDiv', data, layout); }
