--- name: plot3 plot_url: https://plot.ly/~jackp/9205/ arrangement: horizontal language: matlab suite: 3d-line order: 0 sitemap: false --- [X1,Y1]= meshgrid(-5:.2:5,-5:.2:5); syms x y f=((x^2-1)+(y^2-4)+(x^2-1)*(y^2-4))/(x^2+y^2+1)^2 zfun = @(x, y) eval(vectorize(f)) Z1=zfun(X1,Y1); plot3(X1,Y1,Z1) fig2plotly()
