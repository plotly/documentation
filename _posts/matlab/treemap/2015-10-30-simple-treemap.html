--- name: Treemap with Defined Labels and Colors plot_url: https://plot.ly/~chelsea_lyn/8798 arrangement: horizontal language: matlab suite: treemaps order: 0 sitemap: false --- %% Download Treemap here: %% http://www.mathworks.com/matlabcentral/fileexchange/17192-treemap % Custom colors and labels n = 15; data = rand(1,n); colors = (jet(n)+1)/2; % Add labels labels = {}; for i = 1:n labels{i} = sprintf('%2.1f%%',100*data(i)/sum(data)); end fig=figure; cla rectangles = treemap(data); plotRectangles(rectangles,labels,colors) fig2plotly(fig, 'filename', 'simple-treemap')
