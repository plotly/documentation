# Embed Logos in Plotly Graphs with R



# Embedding Logos on a Line Graph


```r
library(plotly)

plot_ly(x = c(1, 2, 3), y = c(1, 2, 3)) %>%
  add_lines() %>%
  layout(
    images = list(
      list(source = "https://images.plot.ly/language-icons/api-home/python-logo.png",
           xref = "paper",
           yref = "paper",
           x= 0,
           y= 1,
           sizex = 0.2,
           sizey = 0.2,
           opacity = 0.8
      ),
      
      list(source = "https://images.plot.ly/language-icons/api-home/matlab-logo.png",
           xref = "x",
           yref = "paper",
           x = 2.7,
           y = 0.25,
           sizex = 0.4,
           sizey = 0.8,
           opacity = 0.8
      ),
      
      list(source =  "https://images.plot.ly/language-icons/api-home/r-logo.png",
           xref = "x",
           yref = "y",
           x = 0.9,
           y = 3.1,
           sizex = 2,
           sizey = 2,
           sizing = "stretch",
           opacity = 0.4,
           layer = "below"
      )
    )
  )
```
