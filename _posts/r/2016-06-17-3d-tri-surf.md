# Basic Tri-Surf Plot

```r
library(plotly)

x = c(0, 1, 2, 0)
y = c(0, 0, 1, 2)
z = c(0, 2, 0, 1)
i = c(0, 0, 0, 1)
j = c(1, 2, 3, 2)
k = c(2, 3, 1, 3)
facecolor = list(
  'rgb(50, 255, 200)',
  'rgb(100, 255, 255)',
  'rgb(150, 255, 115)',
  'rgb(200, 255, 50)'
  )
)

p <- plot_ly(x = x, y = y, z = z,
             i = i, j = j, k = k,
             type = "mesh3d",
             facecolor = facecolor,
             showscale = TRUE
)
p
```

# Cube with Different Face Colors

```r
library(plotly)

x = c(0, 0, 1, 1, 0, 0, 1, 1)
y = c(0, 1, 1, 0, 0, 1, 1, 0)
z = c(0, 0, 0, 0, 1, 1, 1, 1)
i = c(7, 0, 0, 0, 4, 4, 2, 6, 4, 0, 3, 7)
j = c(3, 4, 1, 2, 5, 6, 5, 5, 0, 1, 2, 2)
k = c(0, 7, 2, 3, 6, 7, 1, 2, 5, 5, 7, 6)

facecolor = list(
  'rgb(50, 200, 200)',
  'rgb(100, 200, 255)',
  'rgb(150, 200, 115)',
  'rgb(200, 200, 50)',
  'rgb(230, 200, 10)',
  'rgb(255, 140, 0)'
)


p <- plot_ly(x = x, y = y, z = z,
             i = i, j = j, k = k,
             facecolor = rep(facecolor, each = 2),
             type = "mesh3d")
p
```
