--- name: 3D surface plot plot_url: https://plot.ly/~pbugnion/518 language: scala suite: surface order: 1 sitemap: false arrangement: horizontal --- val xs = (-3.0 to 3.0 by 0.1).toVector val ys = (-3.0 to 3.0 by 0.1).toVector def gaussian2D(x: Double, y: Double) = Math.exp(-x*x - y*y) val zs = xs.map { x => ys.map { y => gaussian2D(x, y) } } val p = ThreeDPlot().withSurface(xs, ys, zs) draw(p, "gaussian-surfaces")
