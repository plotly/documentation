--- name: Subplots in a row plot_url: https://plot.ly/~pbugnion/479 language: scala suite: subplots order: 1 sitemap: false arrangement: horizontal --- import co.theasi.plotly import util.Random val xsLeft = (0 to 100).map { i => Random.nextGaussian } val ysLeft = (0 to 100).map { i => Random.nextGaussian } val xsRight = (0 to 100).map { i => Random.nextDouble } val ysRight = (0 to 100).map { i => Random.nextDouble } val figure = RowFigure(2) .plot(0) { // left-hand plot Plot().withScatter( xsLeft, ysLeft, ScatterOptions().mode(ScatterMode.Marker).name("left")) } .plot(1) { // right-hand plot Plot().withScatter( xsLeft, ysLeft, ScatterOptions().mode(ScatterMode.Marker).name("right")) } draw(figure, "make-subplots", writer.FileOptions(overwrite=true))
