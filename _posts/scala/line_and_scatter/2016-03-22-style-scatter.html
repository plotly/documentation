--- name: Style scatter plots language: scala suite: line_and_scatter plot_url: https://plot.ly/~pbugnion/552 order: 3 sitemap: false arrangement: horizontal --- import co.theasi.plotly import util.Random val n = 500 val xs = (0 until n).map { i => Random.nextDouble } val ys0 = (0 until n).map { i => Random.nextDouble + 2.0 } val ys1 = (0 until n).map { i => Random.nextDouble - 2.0 } val p = Plot() .withScatter(xs, ys0, ScatterOptions() .mode(ScatterMode.Marker) .name("Above") .marker( MarkerOptions() .size(10) .color(152, 0, 0, 0.8) .lineWidth(2) .lineColor(0, 0, 0))) .withScatter(xs, ys1, ScatterOptions() .mode(ScatterMode.Marker) .name("Below") .marker( MarkerOptions() .size(10) .color(255, 182, 193, 0.9) .lineWidth(2))) draw(p, "styled-scatter", writer.FileOptions(overwrite=true))
