--- name: making maps with R plot_url: https://plot.ly/~MattSundquist/19399/lat-vs-long/ arrangement: horizontal language: ggplot2 suite: hover-events order: 1 sitemap: false --- library(plotly) data(canada.cities, package="maps") viz <- ggplot(canada.cities, aes(long, lat)) + borders(regions="canada", name="borders") + coord_equal() + geom_point(aes(text=name, size=pop), colour="red", alpha=1/2, name="cities") ggplotly() ggplotly(filename="r-docs/canada-bubble")
