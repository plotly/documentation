--- name: group plot_url: https://plot.ly/~riddhiman/124/group-specific-line-chart/ arrangement: horizontal language: ggplot2 suite: aes description: Inspired by ggplot2 documentation. order: 1 sitemap: false --- library(plotly) x1 <- 1:100 x2 <- x1 + 100 x3 <- x2 + 100 x <- c(x1, x2, x3) y <- c(2*x1, 5*x2, -2*x3) group <- c(rep("A", length(x1)), rep("B", length(x2)), rep("C", length(x3))) ds <- data.frame(x, y, group) # Use the group aesthetic to ensure lines are drawn separately for each group p <- ggplot(ds, aes(x, y)) + geom_line(aes(group = group, color = group), size = 2) + ggtitle("Group specific line chart") p <- ggplotly(p) p
