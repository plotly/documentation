--- name: shape plot_url: https://plot.ly/~riddhiman/130/groupwise-shapes-and-line-types/ arrangement: horizontal language: ggplot2 suite: aes description: Inspired by ggplot2 documentation. order: 4 sitemap: false --- library(plotly) ds <- data.frame(x = letters[1:5], y = rnorm(20), group = LETTERS[1:4]) # Use aes shape to map individual points and or different groups to different shapes p <- ggplot(ds, aes(x, y)) + geom_point(aes(color = group, shape = group), size = 5) + geom_line(aes(group = group, linetype = group)) + ggtitle("Groupwise shapes and line types") p <- ggplotly(p) p
