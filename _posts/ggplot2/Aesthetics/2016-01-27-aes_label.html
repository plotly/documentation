--- name: label plot_url: https://plot.ly/~riddhiman/128/annotation-with-labels/ arrangement: horizontal language: ggplot2 suite: aes description: Inspired by ggplot2 documentation. order: 3 sitemap: false --- library(plotly) ds <- data.frame(x = rnorm(10), y = rnorm(10), group = LETTERS[1:2]) p <- ggplot(ds, aes(x, y)) + geom_point(aes(color = group), size = 7) + geom_text(aes(label = group), size = 10) + ggtitle("Annotation with labels") p <- ggplotly(p) p
