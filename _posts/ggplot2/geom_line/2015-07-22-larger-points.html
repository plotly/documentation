--- name: geom_line and larger geom_point points plot_url: https://plot.ly/~MattSundquist/17771/total-bill-vs-time/ language: ggplot2 suite: geom_line description: Inspired by the Cookbook for R. order: 2 sitemap: false --- library(plotly) dat1 <- data.frame( sex = factor(c("Female","Female","Male","Male")), time = factor(c("Lunch","Dinner","Lunch","Dinner"), levels=c("Lunch","Dinner")), total_bill = c(13.53, 16.81, 16.24, 17.42) ) # Map sex to different point shape, and use larger points ggplot(data=dat1, aes(x=time, y=total_bill, group=sex, shape=sex)) + geom_line() + geom_point() ggplotly()
