--- name: geom_line with dashed and solid lines plot_url: https://plot.ly/~MattSundquist/17657/mean-vs-x/ language: ggplot2 suite: geom_line description: Inspired by Stack Overflow. order: 10 sitemap: false --- # Learn about API authentication here: https://plot.ly/ggplot2/getting-started # Find your api_key here: https://plot.ly/settings/api library(plotly) x <- c(10, 20, 50, 10, 20, 50) mean = c(52.4, 98.2, 97.9, 74.1, 98.1, 97.6) group = c(1, 1, 1, 2,2,2) upper = c(13.64, 89, 86.4, 13.64, 89, 86.4) lower = c(95.4, 99.8, 99.7, 95.4, 99.8, 99.7) data <- data.frame(x=x,y=mean, group, upper, lower) ggplot(data, aes(x = x, y= mean, group = as.factor(data$group), colour=as.factor(data$group))) + geom_line() + geom_point() + geom_line(aes(y=lower),linetype="dotted") + geom_line(aes(y=upper),linetype="dotted")+ scale_color_manual(name="Groups",values=c("red", "blue"))+ guides(colour = guide_legend(override.aes = list(linetype = 1))) ggplotly()
