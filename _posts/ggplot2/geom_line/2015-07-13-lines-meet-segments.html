--- name: geom_line and geom_segment plot_url: https://plot.ly/~MattSundquist/17373/ language: ggplot2 suite: geom_line description: Inspired by Stack Overflow. order: 12 sitemap: false --- # Learn about API authentication here: https://plot.ly/ggplot2/getting-started # Find your api_key here: https://plot.ly/settings/api library(plotly) x <- rep(1:10, 2) y <- c(1:10, 1:10+5) fac <- gl(2, 10) df <- data.frame(x=x, y=y, fac=fac) ggplot(df, aes(x=x, y=y, linetype=fac)) + geom_line() + geom_segment(aes(x=2, y=7, xend=7, yend=7), colour="red") + scale_linetype_discrete(guide=guide_legend(override.aes=aes(colour="blue"))) ggplotly()
