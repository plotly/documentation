--- name: geom_line with multiple lines plot_url: https://plot.ly/~MattSundquist/17768/total-bill-vs-time/ language: ggplot2 suite: geom_line description: Inspired by the Cookbook for R. order: 1 sitemap: false --- # Learn about API authentication here: https://plot.ly/ggplot2/getting-started # Find your api_key here: https://plot.ly/settings/api library(plotly) dat1 <- data.frame( sex = factor(c("Female","Female","Male","Male")), time = factor(c("Lunch","Dinner","Lunch","Dinner"), levels=c("Lunch","Dinner")), total_bill = c(13.53, 16.81, 16.24, 17.42) ) ggplot(data=dat1, aes(x=time, y=total_bill, group=sex)) + geom_line() + geom_point() py$ggplotly()
