--- name: histograms with geom_vline means plot_url: https://plot.ly/~MattSundquist/17327 language: ggplot2 suite: geom_abline description: Inspired by the Cookbook for R. order: 9 sitemap: false --- # Learn about API authentication here: https://plot.ly/ggplot2/getting-started # Find your api_key here: https://plot.ly/settings/api library(plotly) library(plyr) cdat <- ddply(dat, "cond", summarise, rating.mean=mean(rating)) # With mean lines ggplot(dat, aes(x=rating)) + geom_histogram(binwidth=.5, colour="black", fill="white") + facet_grid(cond ~ .) + geom_vline(data=cdat, aes(xintercept=rating.mean), linetype="dashed", size=1, colour="red") ggplotly()
