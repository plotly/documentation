--- name: overlaid histograms with geom_vline plot_url: https://plot.ly/~MattSundquist/17312/ arrangement: horizontal language: ggplot2 suite: geom_abline description: Inspired by the Cookbook for R. order: 7 sitemap: false --- library(plotly) library(plyr) cdat <- ddply(dat, "cond", summarise, rating.mean=mean(rating)) # Overlaid histograms with means ggplot(dat, aes(x=rating, fill=cond)) + geom_histogram(binwidth=.5, alpha=.5, position="identity") + geom_vline(data=cdat, aes(xintercept=rating.mean), linetype="dashed", size=1) ggplotly()
