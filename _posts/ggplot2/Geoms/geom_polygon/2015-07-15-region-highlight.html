--- name: highlight areas of a plot with geom_polygon plot_url: https://plot.ly/~MattSundquist/18930/wt-vs-hp/ arrangement: horizontal language: ggplot2 suite: geom_polygon description: Inspired by Stack Overflow. order: 2 sitemap: false --- # Learn about API authentication here: https://plot.ly/r/getting-started # Find your api_key here: https://plot.ly/settings/api library(plotly) tmp <- with(mtcars, data.frame(x=c(0, 0, max(wt)*35), y=c(0, max(wt), max(wt)))) ggplot(mtcars, aes(hp, wt)) + geom_polygon(data=tmp, aes(x, y), fill="#d8161688") + geom_point() ggplotly(filename="r-docs/highlight-areas")
