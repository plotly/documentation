--- name: add conditional density curves to plot plot_url: https://plot.ly/~MattSundquist/18766/dim2-vs-dim1/ arrangement: horizontal language: ggplot2 suite: density description: Inspired by Stack Overflow. order: 2 sitemap: false --- library(plotly) set.seed(123) dim1 <- c(rnorm(100, mean=1), rnorm(100, mean=4)) dim2 <- rnorm(200, mean=1) cat <- factor(c(rep("a", 100), rep("b", 100))) mydf <- data.frame(cbind(dim2, dim1, cat)) p <- ggplot(data=mydf, aes(x=dim1, y=dim2, colour=as.factor(cat))) + geom_point() + stat_density(aes(x=dim1, y=(-2+(..scaled..))), position="identity", geom="line") stuff <- ggplot_build(p) xrange <- stuff[[2]]$ranges[[1]]$x.range # extract the x range, to make the new densities align with y-axis ## Get densities of dim2 ds <- do.call(rbind, lapply(unique(mydf$cat), function(lev) { dens <- with(mydf, density(dim2[cat==lev])) data.frame(x=dens$y+xrange[1], y=dens$x, cat=lev) })) p + geom_path(data=ds, aes(x=x, y=y, color=factor(cat))) ggplotly(filename="r-docs/conditional-density-curves")
