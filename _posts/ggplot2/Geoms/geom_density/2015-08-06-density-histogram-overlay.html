--- name: Density and histogram overlay using geom_density plot_url: https://plot.ly/~riddhiman/224/density-with-histogram-overlay/ arrangement: horizontal language: ggplot2 suite: density description: Inspired by ggplot2 documentation. order: 7 sitemap: false --- library(plotly) set.seed(123) df <- data.frame(x <- rchisq(1000, 5, 10), group <- sample(LETTERS[1:5], size = 1000, replace = T)) p <- ggplot(df, aes(x)) + geom_histogram(aes(y = ..density..), alpha = 0.7, fill = "#333333") + geom_density(fill = "#ff4d4d", alpha = 0.5) + theme(panel.background = element_rect(fill = '#ffffff')) + ggtitle("Density with Histogram overlay") ggplotly(p)
