--- name: Density and scatterplot overlay using geom_density plot_url: https://plot.ly/~riddhiman/226/_2d-density-plot-with-scatterplot-overlay/ arrangement: horizontal language: ggplot2 suite: density description: Inspired by ggplot2 documentation. order: 8 sitemap: false --- library(plotly) set.seed(123) df <- data.frame(x <- rchisq(1000, 10, 10), y <- rnorm(1000)) p <- ggplot(df, aes(x, y)) + geom_point(alpha = 0.5) + geom_density_2d() + theme(panel.background = element_rect(fill = '#ffffff')) + ggtitle("2D density plot with scatterplot overlay") ggplotly(p)
