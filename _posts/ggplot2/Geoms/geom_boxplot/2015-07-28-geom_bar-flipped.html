--- name: geom_boxplot flipped plot_url: https://plot.ly/~MattSundquist/18187/cond-vs-rating/ arrangement: horizontal language: ggplot2 suite: box description: Inspired by Cookbook for R. order: 3 sitemap: false --- library(plotly) set.seed(1234) dat <- data.frame(cond = factor(rep(c("A","B"), each=200)), rating = c(rnorm(200),rnorm(200, mean=.8))) ggplot(dat, aes(x=cond, y=rating, fill=cond)) + geom_boxplot() + guides(fill=FALSE) + coord_flip() ggplotly()
