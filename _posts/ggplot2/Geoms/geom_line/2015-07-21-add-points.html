--- name: add points to geom_line plot_url: https://plot.ly/~MattSundquist/17623/y-vs-x/ arrangement: horizontal language: ggplot2 suite: geom_line description: Inspired by Stack Overflow. order: 8 sitemap: false --- library(plotly) library(data.table) d=data.table(x=seq(0, 100, by=0.1), y=seq(0,1000)) ggplot(d, aes(x=x, y=y))+geom_line() #Change the length parameter for fewer or more points thinned <- floor(seq(from=1,to=dim(d)[1],length=70)) ggplot(d, aes(x=x, y=y))+geom_line()+geom_point(data=d[thinned,],aes(x=x,y=y)) ggplotly()
