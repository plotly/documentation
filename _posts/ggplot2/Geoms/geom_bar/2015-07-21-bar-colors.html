--- name: using geom_bar to manually specify colors plot_url: https://plot.ly/~riddhiman/190/specify-manual-colors-in-bar-charts/ arrangement: horizontal language: ggplot2 suite: bar description: Inspired by ggplot2 documentation order: 15 sitemap: false --- library(plotly) library(RColorBrewer) set.seed(123) df <- diamonds[sample(1:nrow(diamonds), size = 1000),] # Simply use fill = a vector of colors p <- ggplot(df, aes(x = color)) + geom_bar(fill = brewer.pal(length(unique(df$color)), "Set3")) + ylab("Count") + ggtitle("Specify manual colors in a bar chart") ggplotly(p)
