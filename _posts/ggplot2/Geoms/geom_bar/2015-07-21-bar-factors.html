--- name: Re-ordering bars shown using geom_bar plot_url: https://plot.ly/~riddhiman/194/bar-chart-with-changed-factor-levels/ arrangement: horizontal language: ggplot2 suite: bar description: Inspired by ggplot2 documentation order: 16 sitemap: false --- library(plotly) df <- data.frame(x = as.factor(LETTERS[1:5]), y = sample(10:20, size = 5)) # First change factor levels df$x <- factor(df$x, levels = c("C", "B", "A", "D", "E")) # Plot p <- ggplot(df, aes(x, y, fill = x)) + geom_bar(stat = "identity") + ggtitle("Bar Chart with changed factor levels") ggplotly(p)
