--- name: geom_histogram with geom_facet plot_url: https://plot.ly/~MattSundquist/18428/a-a-b-b/ arrangement: horizontal language: ggplot2 suite: histogram description: Inspired by Stack Overflow. order: 3 sitemap: false --- library(plotly) df <- data.frame (type=rep(1:2, each=1000), subtype=rep(c("a","b"), each=500), value=rnorm(4000, 0,1)) library(plyr) df.text<-ddply(df,.(type,subtype),summarise,mean.value=mean(value)) ggplot(df, aes(x=value, fill=subtype)) + geom_histogram(position="identity", alpha=0.4)+ facet_grid(. ~ type) ggplotly()
