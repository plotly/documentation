--- name: Adding errorbars to plot using geom_jitter plot_url: https://plot.ly/~RPlotBot/2937/vs/ arrangement: horizontal language: ggplot2 suite: geom_jitter description: Inspired by ggplot2 documentation. order: 5 sitemap: false --- library(plotly) library(dplyr) df <- mtcars %>% group_by(cyl) %>% summarise(min = min(mpg), max = max(mpg)) p <- ggplot(mtcars, aes(x = factor(cyl), y = mpg)) + geom_jitter() + geom_errorbar(data = df, aes(x = factor(cyl), y = 1, ymin = min, ymax = max)) ggplotly(p)
