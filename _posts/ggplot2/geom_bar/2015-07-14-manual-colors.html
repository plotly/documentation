--- name: geom_bar with manual colors plot_url: https://plot.ly/~MattSundquist/17608/total-bill-vs-time/ language: ggplot2 suite: bar description: Inspired by Cookbook for R. order: 7 sitemap: false --- library(plotly) dat1 <- data.frame( sex = factor(c("Female","Female","Male","Male")), time = factor(c("Lunch","Dinner","Lunch","Dinner"), levels=c("Lunch","Dinner")), total_bill = c(13.53, 16.81, 16.24, 17.42) ) ggplot(data=dat1, aes(x=time, y=total_bill, fill=sex)) + geom_bar(stat="identity", position=position_dodge(), colour="black") + scale_fill_manual(values=c("#999999", "#E69F00")) ggplotly()
