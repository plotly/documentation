--- name: ordering variable in geom_bar plot_url: https://plot.ly/~MattSundquist/17678/y-vs-x/ language: ggplot2 suite: bar description: Inspired by Stack Overflow. order: 11 sitemap: false --- # Learn about API authentication here: https://plot.ly/ggplot2/getting-started # Find your api_key here: https://plot.ly/settings/api library(plotly) library(plyr) dane<-data.frame(x=1:10,y=seq(-5,4),g=rep(c('A','B'),each=5)) dane$x<-as.factor(dane$x) ggplot(data=dane,aes(x=x,y=y,fill=g)) + geom_bar(subset=.(g=='A'),stat='identity') + geom_bar(subset=.(g=='B'),stat='identity') + scale_x_discrete(drop=FALSE) ggplotly()
