--- name: filled bar chart with geom_bar plot_url: https://plot.ly/~MattSundquist/16828 language: ggplot2 suite: bar description: Inspired by Cookbook for R. order: 2 sitemap: false --- library(plotly) dat <- data.frame( time = factor(c("Lunch","Dinner"), levels=c("Lunch","Dinner")), total_bill = c(14.89, 17.23) ) ggplot(data=dat, aes(x=time, y=total_bill, fill=time)) + geom_bar(stat="identity") ggplotly()
