--- name: adding margins of error in chart in R plot_url: https://plot.ly/~MattSundquist/19831/population-vs-year/ language: ggplot2 suite: errorbar description: Inspired by Stack Overflow. order: 2 sitemap: false --- library(plotly) population <- data.frame(Year=seq(1790, 1970, length.out=length(uspop)), Population=uspop, Error=rnorm(length(uspop), 5)) library(ggplot2) ggplot(population, aes(x=Year, y=Population, ymin=Population-Error, ymax=Population+Error))+ geom_line()+ geom_point(pch=2)+ geom_errorbar(width=0.9) ggplotly(filename="ggplot2-docs/geom_point")
