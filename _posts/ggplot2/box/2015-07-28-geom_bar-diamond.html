--- name: geom_boxplot with mean using stat_summary plot_url: https://plot.ly/~MattSundquist/18190/rating-vs-cond/ language: ggplot2 suite: box description: Inspired by Cookbook for R. order: 4 sitemap: false --- # Learn about API authentication here: https://plot.ly/ggplot2/getting-started # Find your api_key here: https://plot.ly/settings/api library(plotly) set.seed(1234) dat <- data.frame(cond = factor(rep(c("A","B"), each=200)), rating = c(rnorm(200),rnorm(200, mean=.8))) ggplot(dat, aes(x=cond, y=rating)) + geom_boxplot() + stat_summary(fun.y=mean, geom="point", shape=5, size=4) py$ggplotly()
