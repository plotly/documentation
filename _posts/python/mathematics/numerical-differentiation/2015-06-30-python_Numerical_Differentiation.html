Differentiate the Sine Function¶

How to use numerical differentiation to plot the derivative of a sine wave

In [2]:
# plot a sine wave
x = np.linspace(0, 2*np.pi, 100)
y = np.sin(x)

# use numerical differentiation to compute the derivative plot 
dy = np.zeros(y.shape,np.float)
dy[0:-1] = np.diff(y)/np.diff(x)
dy[-1] = (y[-1] - y[-2])/(x[-1] - x[-2])

trace1 = go.Scatter(
    x=x,
    y=y,
    mode='line',
    name='sin(x)'
)

trace2 = go.Scatter(
    x=x,
    y=dy,
    mode='line',
    name='numerical derivative of sin(x)'
)

trace_data = [trace1, trace2]
py.iplot(trace_data, filename='Numerical Differentiation')
