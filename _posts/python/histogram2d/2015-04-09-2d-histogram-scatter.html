--- name: 2D Histogram Overlaid with a Scatter Chart plot_url: https://plot.ly/~jackp/9363/ language: python suite: histogram2d order: 2 sitemap: false --- # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api import plotly.plotly as py from plotly.graph_objs import * import numpy as np x0 = np.random.randn(100)/5. + 0.5 # 5. enforces float division y0 = np.random.randn(100)/5. + 0.5 x1 = np.random.rand(50) y1 = np.random.rand(50) + 1.0 x = np.concatenate([x0, x1]) y = np.concatenate([y0, y1]) trace1 = Scatter( x=x0, y=y0, mode='markers', showlegend=False, marker=Marker( symbol='x', opacity=0.7, color='white', size=8, line=Line(width=1), ) ) trace2 = Scatter( x=x1, y=y1, mode='markers', showlegend=False, marker=Marker( symbol='circle', opacity=0.7, color='white', size=8, line=Line(width=1), ) ) trace3 = Histogram2d( x=x, y=y, colorscale='YIGnBu', zmax=10, nbinsx=14, nbinsy=14, zauto=False, ) layout = Layout( xaxis=XAxis( ticks='', showgrid=False, zeroline=False, nticks=20 ), yaxis=YAxis( ticks='', showgrid=False, zeroline=False, nticks=20 ), autosize=False, height=550, width=550, hovermode='closest', ) data = Data([trace1, trace2, trace3]) fig = Figure(data=data,layout=layout) url = py.plot(fig, filename='2d-histogram-scatter')
