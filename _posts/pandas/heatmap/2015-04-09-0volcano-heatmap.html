--- name: Basic Heatmap in Pandas plot_url: https://plot.ly/~jackp/10549/ language: pandas suite: heatmap order: 0 sitemap: false arrangement: horizontal --- # Learn about API authentication here: https://plot.ly/pandas/getting-started # Find your api_key here: https://plot.ly/settings/api import plotly.plotly as py import plotly.graph_objs as go import pandas as pd df = pd.read_csv('https://raw.githubusercontent.com/plotly/datasets/master/volcano.csv') data = [go.Heatmap( z=df.values.tolist(), colorscale='Viridis')] py.iplot(data, filename='pandas-heatmap')
