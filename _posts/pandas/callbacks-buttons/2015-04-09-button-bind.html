--- name: Bind button events to Pandas charts in JavaScript plot_url: https://codepen.io/plotly/embed/OVrbVY/?height=770&theme-id=15263&default-tab=result height: 850 language: pandas suite: button-events arrangement: horizontal sitemap: false order: 0 --- import plotly.plotly as py import plotly.graph_objs as go import numpy as np size = 100 x = np.linspace(-2*np.pi, 2*np.pi, size) y = np.linspace(-2*np.pi, 2*np.pi, size) z = np.empty((size, size)) for i, xi in enumerate(x): for j, yj in enumerate(y): r2 = (xi**2+yj**2) z[i][j] = np.sin(xi)*np.cos(yj)*np.sin(r2)/(np.log(r2+1)) data = [go.Contour(z=z, x=x, y=y)] plot_url = py.plot(data, filename='simple-contour')
