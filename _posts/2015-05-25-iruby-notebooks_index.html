Creating an interactive Ruby/Plotly graph inside an IPython/Jupyter notebook


require 'plotly'

plotly = PlotLy.new('username', 'api_key')

data = { 
  x: ['2013-10-04 22:23:00', '2013-11-04 22:23:00', '2013-12-04 22:23:00'],
  y: [1, 3, -9]
}

args = {
  filename: 'ruby_test_time_series',
  fileopt: 'overwrite',
  style: { type: 'scatter' },
  layout: {
    title: 'Ruby Test 22'
  },
  world_readable: true
}

plotly.plot(data, args) do |response|
  puts response['url']
end

# prints the plotly url ( e.g. https://plot.ly/~Ruby/6 )
IRuby.html '<iframe src="https://plot.ly/~Ruby/6" width=700 height=350></iframe>'
