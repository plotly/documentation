--- name: Setting Title Font Size and Font Family plot_url: https://plot.ly/~tarzzz/1230 arrangement: horizontal language: matplotlib suite: labels order: 1 sitemap: false --- import matplotlib.pyplot as plt import numpy as np import plotly.plotly as py import plotly.tools as tls # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api x = np.array(range(1,13)) y = np.random.randint(100, size=x.shape) plt.plot(x,y) plt.title('The title with font size: 20, and font:monospace') plt.tight_layout() fig = plt.gcf() plotly_fig = tls.mpl_to_plotly( fig ) plotly_fig['layout']['titlefont'].update({'color':'blue', 'size':20, 'family':'monospace'}) plot_url = py.plot(plotly_fig)
