--- name: Annotations with Arrow plot_url: https://plot.ly/~tarzzz/974 arrangement: horizontal language: matplotlib suite: annotations order: 2 sitemap: false description: Inspired by Matplotlib Docs --- import matplotlib.pyplot as plt import numpy as np import plotly.plotly as py import plotly.tools as tls # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api t = np.arange(-5.0, 5.0, 0.01) s = np.sin(4 * np.pi * t) * np.exp(-5 * np.absolute(t)) mpl_fig = plt.figure() ax = mpl_fig.add_subplot(111) line, = ax.plot(t, s, lw=2) plt.annotate('Global Maximum', xy=(0.08, 0.58), xytext=(0.08, 0.58), arrowprops=dict(facecolor='black', shrink=0.05)) ax.set_ylim(-2,2) ax.set_xlim(-5,5) plotly_fig = tls.mpl_to_plotly( mpl_fig ) # Annotations are hidden by default.. plotly_fig['layout']["annotations"][0].update({ "showarrow":True }) plot_url = py.plot_mpl(plotly_fig, filename='mpl-annotation-with-arrow')
