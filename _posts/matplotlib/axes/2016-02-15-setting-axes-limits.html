--- name: Setting Axes Fixed Range with Xlim and Ylim plot_url: https://plot.ly/~tarzzz/1310 arrangement: horizontal language: matplotlib suite: axes order: 3 sitemap: false description: Inspired by matplotlib tutorials --- import matplotlib.pyplot as plt import numpy as np import plotly.plotly as py import plotly.tools as tls # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api x = np.arange(1, 7, 0.4) y0 = np.sin(x) y = y0.copy() + 2.5 plt.title('Setting Axis Limits') plt.step(x, y) y -= 0.5 plt.step(x, y) y -= 0.5 plt.step(x, y) plt.xlim(0, 7) plt.ylim(-0.5, 4) plt.xlabel('X Axis limit is (0,7)') plt.ylabel('Y Axis limit is (-0.5,4)') mpl_fig = plt.gcf() plotly_fig = tls.mpl_to_plotly( mpl_fig ) plot_url = py.plot(plotly_fig)
