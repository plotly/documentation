--- name: Matplotlib Twinx - Graph with two Y Axes plot_url: https://plot.ly/~tarzzz/1374 arrangement: horizontal language: matplotlib suite: axes order: 3 sitemap: false --- import matplotlib.pyplot as plt import numpy as np import plotly.plotly as py import plotly.tools as tls # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api x=[1, 2, 3, 4, 5] y=[2, 4, 7, 9, 5] y1=[20, 45, 79, 90, 190] plt.plot(x,y, color="r") plt.plot(x,y1, color="b") fig = plt.gcf() plotly_fig = tls.mpl_to_plotly(fig) plotly_fig['data'][0].update({'name':'YAxis1 Data'}) plotly_fig['data'][1].update({'yaxis':'y2', 'name':'YAxis2 Data'}) plotly_fig['layout'].update(yaxis=dict(title='Y Axis1', titlefont=dict(color='rgb(255, 0, 0)'), tickfont=dict(color='rgb(255, 0, 0)'), side='left') ) plotly_fig['layout'].update(yaxis2=dict(title='Y Axis2', titlefont=dict(color='rgb(0, 0, 255)'), tickfont=dict(color='rgb(0, 0, 255)'), overlaying='y',side='right') ) plotly_fig['layout']['showlegend'] = True plot_url = py.plot(plotly_fig, filename='mpl-two-y-axis')
