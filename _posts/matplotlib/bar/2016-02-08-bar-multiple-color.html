--- name: Bar Chart with Multiple Colors plot_url: https://plot.ly/~tarzzz/1032 arrangement: horizontal language: matplotlib suite: bar order: 8 sitemap: false --- import matplotlib.pyplot as plt import numpy as np import plotly.plotly as py import plotly.tools as tls # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api mpl_fig = plt.figure() ax = mpl_fig.add_subplot(111) N = 5 ind = np.arange(N) width = 0.3 vals = [1,2,3,4,5] colors = ['b','r','g','g','g'] barlist = ax.bar(ind, vals, width) barlist[0].set_color('g') barlist[1].set_color('b') plotly_fig = tls.mpl_to_plotly( mpl_fig ) plotly_fig["data"][0]["marker"]["color"] = ["rgba(255,0,0,0.5)", "rgba(0,255,0,0.5)", "rgba(0,0,0,255,0.5)", "rgba(122,122,122,0.7)", "rgba(0,122,122,0.5"] plot_url = py.plot(plotly_fig, filename='bars-with-multiple-colors')
