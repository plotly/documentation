--- name: log bar plot plot_url: https://plot.ly/~MattSundquist/19281/ arrangement: horizontal language: matplotlib suite: bar description: Inspired by matplotlib docs. sitemap: false --- # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api from matplotlib import pylab import plotly.plotly as py log = plt.figure() data = ((3,1000), (10,3), (100,30), (500, 800), (50,1)) pylab.xlabel("FOO") pylab.ylabel("FOO") pylab.title("Testing") pylab.gca().set_yscale('log') dim = len(data[0]) w = 0.75 dimw = w / dim x = pylab.arange(len(data)) for i in range(len(data[0])) : y = [d[i] for d in data] b = pylab.bar(x + i * dimw, y, dimw, bottom=0.001) pylab.gca().set_xticks(x + w / 2) pylab.gca().set_ylim( (0.001,1000)) plot_url = py.plot_mpl(log, filename='mpl-log')
