--- name: Matplotlib Colormap With Legend plot_url: https://plot.ly/~tarzzz/1940/ arrangement: horizontal language: matplotlib suite: colors_and_colormaps order: 5 sitemap: false --- import plotly.plotly as py import plotly.graph_objs as go import plotly.tools as tls # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api import matplotlib.pyplot as plt import numpy as np fig = plt.figure() ax = fig.add_subplot(111) for i in range(10): x = np.random.normal(loc=i, size=100) y = np.random.normal(loc=i, size=100) ax.scatter(x,y,c=y, cmap=plt.cm.get_cmap('RdBu'), label='Trace {}'.format(i)) plotly_fig = tls.mpl_to_plotly(fig) plotly_fig['layout']['showlegend'] = True py.iplot(plotly_fig)
