--- name: Box Plot with Outliers plot_url: https://plot.ly/~tarzzz/1898 arrangement: horizontal language: matplotlib suite: box order: 3 sitemap: false --- import matplotlib.pyplot as plt import numpy as np import plotly.plotly as py import plotly.tools as tls # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api spread = np.random.rand(50) * 100 center = np.ones(25) * 50 flier_high = np.random.rand(10) * 100 + 100 flier_low = np.random.rand(10) * -100 data = np.concatenate((spread, center, flier_high, flier_low), 0) # change outlier point symbols mpl_fig = plt.figure() plt.boxplot(data, 0, 'gD') plotly_fig = tls.mpl_to_plotly(mpl_fig) plotly_url = py.plot(plotly_fig)
