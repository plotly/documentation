--- name: Multiple Box Plots plot_url: https://plot.ly/~tarzzz/1204 arrangement: horizontal language: matplotlib suite: box order: 2 sitemap: false --- import matplotlib.pyplot as plt import numpy as np import plotly.plotly as py import plotly.tools as tls # Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api # Generating sample data N = 1000 data_set1 = np.random.normal(1, 1, N) data_set2 = np.random.lognormal(1, 1, N) data_set3 = np.random.exponential(1, N) data_set4 = np.random.gumbel(6, 4, N) data_set5 = np.random.triangular(2, 9, 11, N) data = [data_set1, data_set2, data_set3, data_set4, data_set5] mpl_fig = plt.figure() ax = mpl_fig.add_subplot(111) ax.boxplot(data) plotly_fig = tls.mpl_to_plotly( mpl_fig ) plot_url = py.plot(plotly_fig, 'mpl-multiple-boxplot')
