Simple Inset Graph

# Learn about API authentication here: https://plot.ly/julia/getting-started
# Find your api_key here: https://plot.ly/settings/api

using Plotly


trace1 = [
  "x" => [1, 2, 3],
  "y" => [4, 3, 2],
  "type" => "scatter"
]
trace2 = [
  "x" => [20, 30, 40],
  "y" => [30, 40, 50],
  "xaxis" => "x2",
  "yaxis" => "y2",
  "type" => "scatter"
]
data = [trace1, trace2]
layout = [
  "yaxis2" => [
    "domain" => [0.6, 0.95],
    "anchor" => "x2"
  ],
  "xaxis2" => [
    "domain" => [0.6, 0.95],
    "anchor" => "y2"
  ]
]
response = Plotly.plot(data, ["layout" => layout, "filename" => "simple-inset", "fileopt" => "overwrite"])
plot_url = response["url"]
