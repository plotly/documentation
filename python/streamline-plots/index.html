Streamline Plot Example

# Learn about API authentication here: https://plot.ly/python/getting-started

import plotly.plotly as py
from plotly.tools import FigureFactory as FF

import numpy as np

# Add data
x = np.linspace(-3, 3, 100)
y = np.linspace(-3, 3, 100)
Y, X = np.meshgrid(x, y)
u = -1 - X**2 + Y
v = 1 + X - Y**2

# Create streamline figure
fig = FF.create_streamline(x, y, u, v, arrow_scale=.1)

# Plot
url = py.plot(fig, filename='Streamline Plot Example')

import plotly.plotly as py
from plotly.tools import FigureFactory as FF
import plotly.graph_objs as go

import numpy as np

#  Add data
N = 50
x_start, x_end = -2.0, 2.0
y_start, y_end = -1.0, 1.0
x = np.linspace(x_start, x_end, N)
y = np.linspace(y_start, y_end, N)
X, Y = np.meshgrid(x, y)
source_strength = 5.0
x_source, y_source = -1.0, 0.0

# Compute the velocity field on the mesh grid
u = (source_strength/(2*np.pi) *
     (X-x_source)/((X-x_source)**2 + (Y-y_source)**2))
v = (source_strength/(2*np.pi) *
     (Y-y_source)/((X-x_source)**2 + (Y-y_source)**2))

# Create streamline figure
fig = FF.create_streamline(x, y, u, v,
                           name='streamline')

# Add source point
source_point = go.Scatter(x=[x_source], y=[y_source],
                          mode='markers',
                          marker=Marker(size=14),
                          name='source point')

# Add source point to figure
fig['data'].append(source_point)

url = py.plot(fig, filename='streamline_source')

Streamline Source and Sink Plot

# Learn about API authentication here: https://plot.ly/python/getting-started
# Find your api_key here: https://plot.ly/settings/api

import plotly.plotly as py
from plotly.tools import FigureFactory as FF
import plotly.graph_objs as go

import numpy as np

# Add data
N = 50
x_start, x_end = -2.0, 2.0
y_start, y_end = -1.0, 1.0
x = np.linspace(x_start, x_end, N)
y = np.linspace(y_start, y_end, N)

X, Y = np.meshgrid(x, y)

strength_source = 5.0
x_source, y_source = -1.0, 0.0

# Compute the velocity field
u_source = strength_source/(2*np.pi) * (X-x_source)/((X-x_source)**2 + (Y-y_source)**2)
v_source = strength_source/(2*np.pi) * (Y-y_source)/((X-x_source)**2 + (Y-y_source)**2)

strength_sink = -5.0
x_sink, y_sink = 1.0, 0.0

u_sink = strength_sink/(2*np.pi) * (X-x_sink)/((X-x_sink)**2 + (Y-y_sink)**2)
v_sink = strength_sink/(2*np.pi) * (Y-y_sink)/((X-x_sink)**2 + (Y-y_sink)**2)

u_pair = u_source + u_sink
v_pair = v_source + v_sink

# Make streamline figure
fig = FF.create_streamline(x, y, u_pair, v_pair,
                           arrow_scale=.07, angle=np.pi/9,
                           density=1.5, name='streamline')

# Make source points
points = go.Scatter(x=[x_source, x_sink], y=[y_source, y_sink],
                    mode='markers', marker=dict(size=14),
                    name='source points')

# Add source points to figure
fig['data'].append(points)

# Add title to plot
fig['layout'].update(title='Streamline Source and Sink Plot')

# Plot
url = py.plot(fig, filename='Streamline Source and Sink Plot')
