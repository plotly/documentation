# Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api import plotly.plotly as py from plotly.tools import FigureFactory as FF from plotly.graph_objs import * import numpy as np # Add data x = np.linspace(-3, 3, 100) y = np.linspace(-3, 3, 100) Y, X = np.meshgrid(x, y) u = -1 - X**2 + Y v = 1 + X - Y**2 # Create streamline figure fig = FF.create_streamline(x, y, u, v, arrow_scale=.1) # Plot url = py.plot( fig, filename='Streamline Plot Example' )
