library(plotly) set.seed(1) x <- 1:10 dd <- rbind(data.frame(x=x,fac="a", y=x+rnorm(10)), data.frame(x=2*x,fac="b", y=x+rnorm(10))) coef <- lm(y~x:fac, data=dd)$coefficients qplot(data=dd, x=x, y=y, color=fac)+ geom_abline(slope=coef["x:faca"], intercept=coef["(Intercept)"])+ geom_abline(slope=coef["x:facb"], intercept=coef["(Intercept)"]) ggplotly()
