library(plotly) test <- data.frame(names=rep(c("A","B","C"), 5), values=1:15) test2 <- data.frame(names=c("A","B","C"), values=5:7) df <- data.frame(names=c(paste(test$names), paste(test2$names)), num=c(rep(1, nrow(test)), rep(2, nrow(test2))), values=c(test$values, test2$values)) library(plyr) df2 <- ddply(df, c("names", "num"), summarise, values = sum(values)) ggplot(df2, aes(x=factor(names), y=values, fill=factor(num))) + geom_bar(stat="identity", colour="black", position="dodge")
