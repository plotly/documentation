
var plotly = require('plotly')('TestBot', 'r1neazxo9w')

  y: [4, 3, 2], 
  type: "scatter"
};
var trace2 = {
  x: [20, 30, 40], 
  y: [30, 40, 50], 
  xaxis: "x2", 
  yaxis: "y2", 
  type: "scatter"
};

    anchor: "y2"
  }
