var x = [];
var y = [];

for (var i = 0; i &lt; 500; i ++) {
	x[i] = Math.random();
	y[i] = Math.random() + 1;
}


var plotly = require('plotly')('TestBot', 'r1neazxo9w')
var data = [
  {
    x: x, 
    y: y, 
    type: "histogram2d"
  }
];
var graph_options = {filename: "2d-histogram", fileopt: "overwrite"}
plotly.plot(data, graph_options, function (err, msg) {
    console.log(msg);
});
