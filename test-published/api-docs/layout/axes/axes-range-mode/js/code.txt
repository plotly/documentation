var plotly = require('plotly')('Node.js-Demo-Account', 'dvlqkmw0zm')
var data = [
  {
    x: [2, 4, 6], 
    y: [-3, 0, 3], 
    type: "scatter"
  }
];
var layout = {
  showlegend: false, 
  xaxis: {
    rangemode: "tozero", 
    autorange: true
  }, 
  yaxis: {
    rangemode: "nonnegative", 
    autorange: true
  }
};
var graph_options = {layout: layout, fileopt: "overwrite", filename: "axes-range-mode"}
plotly.plot(data, graph_options, function (err, msg) {
    console.log(msg);
});
